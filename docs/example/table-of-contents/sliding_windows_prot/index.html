    滑动窗口 #

    滑动窗口的净代码量不超过十行，但是这十行废了我两天的功夫还没搞清楚

    class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if(s.length() == 0) {
            return 0;
        }
        int left = 0;
        int right = 0;
        int res = 0;
        vector<int> char_vec(128, -1);
        while(right <= s.length() - 1) {
            char right_char = s[right];
            int index = char_vec[right_char];
            if(index != -1 && index >= left && index <= right) {
                left = index + 1;
            }
            res = max(res, right - left + 1);
            char_vec[right_char] = right;
            ++right;
        }
        return res;
    }
};

    上面是lc上别人已经ac的答案，我们来顺一遍当输入是abcabcbb时是怎么个情况。

    第一轮迭代，right = 0, right_char = a, index是vector中第"a"个所在的值 = -1, res = max(0, 1) = 1, vector中第"a"个所在的值更新为right即为0, right++为1

    第二轮迭代，right = 1, right_char = b, index是vector中第"b"个所在的值 = -1, res = max(1, 2) = 2, vector中第"b"个所在的值更新为right即位1, right++为2

    第三轮，right = 2, right_char = c, index = -1, res = max(2, 3) = 3, ector中第"b"个所在的值更新为right即位2, right++为3

    第四轮，right = 3, right_char = a, index为0, 因为left是0，right是3, left=index+1 = 1, res = max(3, 3) = 3, vec[a] = 3, right = 4
